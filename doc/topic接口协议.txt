1、发表话题
[POST] /app/{appid}/folder/{folderPath}/article
[PARAMS]
ticket:用户票据，用于用户权限验证
article:{
	title:标题
	brief:简介
	tags:[
		{
			name:标签名
		}
	]
	content:{
		size:段落数量
		sections:[
			{
				type:段落类型(text/image/audio/vidio)
				attr:{
					排版样式
				}
				data:{
					段落内容
				}
			}
		]
	}
}
[RESPONSE]
{
	code:200
	article:{
		id:话题ID
		title:标题
		brief:简介
		tags:[
			{
				name:标签名
			}
		]
		content:{
			size:段落数量
			sections:[
				{
					type:段落类型(text/image/audio/vidio)
					attr:{
						排版样式
					}
					data:{
						段落内容
					}
				}
			]
		}
		creator:{
			uid:用户ID
			nick:昵称
			avatar:头像
		}
		utime:更新时间
		ctime:创建时间
	}
}

4、话题列表

[GET]/app/{appid}/folder/{folderPath}/articles
[PARAMS]
limit=返回话题的话量
last=上一次列表最后一项话题
[RESPONSE]
{
	code:200
	page:{
		total:总数
		last:列表最后一项偏移值
		count:当前列表数量
		more:是否存在下一页
	}
	articles:[
		{
			id:话题ID
			title:标题
			brief:简介
			tags:[
				{
					name:标签名
				}
			]
			thumbs:{
				size:缩略图数量
				images:[
					{
						url:图片地址
						width:图片宽度
						height:图片高度
					}
				]
			}
			stats:{
				acclaimCnt:点赞数量
				postCnt:评论数量
			}
			creator:{
				uid:用户ID
				nick:昵称
				avatar:头像
			}
			utime:更新时间
			ctime:创建时间
		}
	]
}
